import polars as pl
from itertools import combinations


def match_amounts_for_date(
    left: pl.DataFrame,
    right: pl.DataFrame,
    tolerance: int = 3,
) -> pl.DataFrame:
    left_rows = left.with_row_index("df1_row_id").to_dicts()
    right_rows = right.with_row_index("df2_row_id").to_dicts()

    used = set()
    matches = []

    # Usually useful: match smaller/exact targets first
    right_rows = sorted(right_rows, key=lambda r: r["amount"])

    for r in right_rows:
        available = [x for x in left_rows if x["df1_row_id"] not in used]

        best_combo = None
        best_score = None
        best_delta = None

        for k in range(1, len(available) + 1):
            for combo in combinations(available, k):
                combo_sum = sum(x["amount"] for x in combo)
                delta = combo_sum - r["amount"]

                if abs(delta) <= tolerance:
                    # Prefer smallest deviation, then fewest rows
                    score = (abs(delta), k)

                    if best_score is None or score < best_score:
                        best_score = score
                        best_combo = combo
                        best_delta = delta

        if best_combo is not None:
            for x in best_combo:
                used.add(x["df1_row_id"])
                matches.append({
                    "date": x["date"],
                    "amount": x["amount"],
                    "credit_id": r["credit_id"],
                    "matched_df2_amount": r["amount"],
                    "matched_sum": sum(y["amount"] for y in best_combo),
                    "delta": best_delta,
                    "abs_delta": abs(best_delta),
                })

    return pl.DataFrame(matches)
