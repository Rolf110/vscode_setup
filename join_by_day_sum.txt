import re
from datetime import date, datetime, timedelta

import polars as pl


def reconcile_amounts_full(
    df1: pl.DataFrame,
    df2: pl.DataFrame,
    *,
    tolerance: int | float = 3,
    date_tolerance: int | float | str | timedelta = "1d",
    date_col: str = "local_date",
    amount_col: str = "amount_rur",
    credit_col: str = "credit_id",
    extra_left_cols: list[str] | None = None,
    extra_right_cols: list[str] | None = None,
    right_prefix: str = "df2_",
    output_cols: list[str] | None = None,
    max_combo_size: int | None = None,  # kept only for backward compatibility
) -> pl.DataFrame:
    """
    Reconcile raw df1 transactions to df2 daily aggregates.

    Matching logic:
        1. df2 is aggregated by date, because it may have multiple rows
           per date for different credit_id values.
        2. df1 is grouped by date logically, but raw rows are preserved.
        3. A df1 date group may match a df2 date group within ±date_tolerance.
        4. Match is accepted when:

               abs(sum(df1.amount_rur) - sum(df2.amount_rur)) <= tolerance

           and

               abs(df1.local_date - df2.local_date) <= date_tolerance

    date_tolerance examples:
        1       -> 1 day
        "1d"    -> 1 day
        "2 days" -> 2 days
        "24h"   -> 1 day
        timedelta(days=1)

    Output:
        local_date          = original df1 date for matched / left_only rows
        matched_df2_date    = matched df2 aggregate date
        date_delta_days     = df1 date - df2 date, in days
        credit_id           = list of credit_ids from matched df2 date
    """

    extra_left_cols = extra_left_cols or []
    extra_right_cols = extra_right_cols or []

    required_left = {date_col, amount_col, *extra_left_cols}
    required_right = {date_col, amount_col, credit_col, *extra_right_cols}

    missing_left = required_left - set(df1.columns)
    missing_right = required_right - set(df2.columns)

    if missing_left:
        raise ValueError(f"df1 is missing columns: {sorted(missing_left)}")
    if missing_right:
        raise ValueError(f"df2 is missing columns: {sorted(missing_right)}")

    def parse_date_tolerance(value: int | float | str | timedelta) -> float:
        if isinstance(value, timedelta):
            return abs(value.total_seconds() / 86_400)

        if isinstance(value, (int, float)):
            return abs(float(value))

        if isinstance(value, str):
            value_clean = value.strip().lower()

            match = re.fullmatch(
                r"([0-9]+(?:\.[0-9]+)?)\s*(d|day|days)",
                value_clean,
            )
            if match:
                return abs(float(match.group(1)))

            match = re.fullmatch(
                r"([0-9]+(?:\.[0-9]+)?)\s*(h|hour|hours)",
                value_clean,
            )
            if match:
                return abs(float(match.group(1)) / 24)

        raise ValueError(
            "date_tolerance must be int/float days, timedelta, "
            'or string like "1d", "2 days", "24h".'
        )

    def normalize_to_date(value) -> date | None:
        if value is None:
            return None

        if isinstance(value, datetime):
            return value.date()

        if isinstance(value, date):
            return value

        if isinstance(value, str):
            value_clean = value.strip()

            try:
                return datetime.fromisoformat(
                    value_clean.replace("Z", "+00:00")
                ).date()
            except ValueError:
                pass

            try:
                return date.fromisoformat(value_clean[:10])
            except ValueError:
                pass

        raise ValueError(
            f"Could not parse {date_col} value as date: {value!r}"
        )

    def date_distance_days(left_date, right_date) -> float:
        left_norm = normalize_to_date(left_date)
        right_norm = normalize_to_date(right_date)

        if left_norm is None and right_norm is None:
            return 0

        if left_norm is None or right_norm is None:
            return float("inf")

        return abs((left_norm - right_norm).days)

    def signed_date_delta_days(left_date, right_date) -> int | None:
        left_norm = normalize_to_date(left_date)
        right_norm = normalize_to_date(right_date)

        if left_norm is None or right_norm is None:
            return None

        return (left_norm - right_norm).days

    max_date_distance = parse_date_tolerance(date_tolerance)

    df1_i = df1.with_row_index("__df1_row_id")

    # Aggregate right table by date.
    # amount_col is summed.
    # credit_col and extra_right_cols are collected as unique lists.
    df2_daily = (
        df2
        .group_by(date_col, maintain_order=True)
        .agg(
            [
                pl.when(pl.col(amount_col).is_not_null().any())
                .then(pl.col(amount_col).sum())
                .otherwise(None)
                .alias(amount_col),

                pl.col(credit_col)
                .drop_nulls()
                .unique(maintain_order=True)
                .alias(credit_col),

                *[
                    pl.col(c)
                    .drop_nulls()
                    .unique(maintain_order=True)
                    .alias(c)
                    for c in extra_right_cols
                ],
            ]
        )
        .with_row_index("__df2_day_id")
    )

    right_extra_out_cols = [f"{right_prefix}{c}" for c in extra_right_cols]

    base_cols = [
        date_col,
        amount_col,
        credit_col,
        "matched_df2_date",
        "matched_df2_amount",
        "match_status",
        "matched_sum",
        "delta",
        "date_delta_days",
    ]

    created_cols = list(
        dict.fromkeys(base_cols + extra_left_cols + right_extra_out_cols)
    )

    def make_record(
        *,
        left_row: dict | None,
        right_row: dict | None,
        status: str,
        amount_value,
        credit_value,
        matched_df2_date,
        matched_df2_amount,
        matched_sum,
        delta,
        date_delta,
    ) -> dict:
        rec = {c: None for c in created_cols}

        rec[date_col] = (
            left_row[date_col]
            if left_row is not None
            else right_row[date_col]
        )

        rec[amount_col] = amount_value
        rec[credit_col] = credit_value
        rec["matched_df2_date"] = matched_df2_date
        rec["matched_df2_amount"] = matched_df2_amount
        rec["match_status"] = status
        rec["matched_sum"] = matched_sum
        rec["delta"] = delta
        rec["date_delta_days"] = date_delta

        for c in extra_left_cols:
            rec[c] = left_row.get(c) if left_row is not None else None

        for c in extra_right_cols:
            rec[f"{right_prefix}{c}"] = (
                right_row.get(c) if right_row is not None else None
            )

        return rec

    def daily_sum(rows: list[dict]):
        values = [
            row[amount_col]
            for row in rows
            if row.get(amount_col) is not None
        ]
        return sum(values) if values else None

    # Preserve original left row order.
    left_rows_all = df1_i.to_dicts()
    right_rows_all = df2_daily.to_dicts()

    left_by_date: dict[object, list[dict]] = {}
    for row in left_rows_all:
        left_by_date.setdefault(row[date_col], []).append(row)

    left_day_rows = [
        {
            "left_date": left_date,
            "left_rows": rows,
            "left_sum": daily_sum(rows),
        }
        for left_date, rows in left_by_date.items()
    ]

    used_left_dates = set()
    used_right_ids = set()
    records = []

    # Greedy one-to-one daily matching:
    # one left date group can match one right date group within date tolerance.
    for left_day in left_day_rows:
        left_date = left_day["left_date"]
        left_rows = left_day["left_rows"]
        left_sum = left_day["left_sum"]

        best_right = None
        best_delta = None
        best_date_delta = None
        best_score = None

        if left_sum is not None:
            for right_row in right_rows_all:
                right_id = right_row["__df2_day_id"]

                if right_id in used_right_ids:
                    continue

                right_amount = right_row.get(amount_col)

                if right_amount is None:
                    continue

                distance = date_distance_days(left_date, right_row[date_col])

                if distance > max_date_distance:
                    continue

                amount_delta = left_sum - right_amount

                if abs(amount_delta) > tolerance:
                    continue

                signed_date_delta = signed_date_delta_days(
                    left_date,
                    right_row[date_col],
                )

                # Prefer:
                # 1. smallest amount deviation
                # 2. closest date
                # 3. deterministic right-side order
                score = (
                    abs(amount_delta),
                    distance,
                    right_id,
                )

                if best_score is None or score < best_score:
                    best_score = score
                    best_right = right_row
                    best_delta = amount_delta
                    best_date_delta = signed_date_delta

        if best_right is not None:
            used_left_dates.add(left_date)
            used_right_ids.add(best_right["__df2_day_id"])

            for left_row in left_rows:
                records.append(
                    make_record(
                        left_row=left_row,
                        right_row=best_right,
                        status="matched",
                        amount_value=left_row[amount_col],
                        credit_value=best_right[credit_col],
                        matched_df2_date=best_right[date_col],
                        matched_df2_amount=best_right[amount_col],
                        matched_sum=left_sum,
                        delta=best_delta,
                        date_delta=best_date_delta,
                    )
                )

    # Unmatched df1 date groups
    for left_day in left_day_rows:
        left_date = left_day["left_date"]

        if left_date in used_left_dates:
            continue

        left_rows = left_day["left_rows"]
        left_sum = left_day["left_sum"]

        for left_row in left_rows:
            records.append(
                make_record(
                    left_row=left_row,
                    right_row=None,
                    status="left_only",
                    amount_value=left_row[amount_col],
                    credit_value=None,
                    matched_df2_date=None,
                    matched_df2_amount=None,
                    matched_sum=left_sum,
                    delta=None,
                    date_delta=None,
                )
            )

    # Unmatched df2 daily aggregates
    for right_row in right_rows_all:
        if right_row["__df2_day_id"] in used_right_ids:
            continue

        records.append(
            make_record(
                left_row=None,
                right_row=right_row,
                status="right_only",
                amount_value=None,
                credit_value=right_row[credit_col],
                matched_df2_date=right_row[date_col],
                matched_df2_amount=right_row[amount_col],
                matched_sum=None,
                delta=None,
                date_delta=None,
            )
        )

    result = (
        pl.DataFrame(records)
        if records
        else pl.DataFrame({c: [] for c in created_cols})
    )

    final_cols = output_cols or created_cols

    for c in final_cols:
        if c not in result.columns:
            result = result.with_columns(pl.lit(None).alias(c))

    return result.select(final_cols)
