import math
import pandas as pd
import plotly.graph_objects as go
import plotly.express as px


def plot_monthly_spending_donuts(
    df: pd.DataFrame,
    category_col: str = "operation type",
    date_col: str = "local_date",
    amount_col: str = "tatal_amount",   # change to "total_amount" if needed
    events_col: str = "n_events",
    max_cols: int = 6,
    hole: float = 0.55,
    use_abs: bool = True,
    show_total_in_center: bool = True,
    title: str | None = "Monthly spending by category",
):
    d = df.copy()

    required = [category_col, date_col, amount_col, events_col]
    missing = [c for c in required if c not in d.columns]
    if missing:
        raise ValueError(f"Missing columns: {missing}")

    d[category_col] = d[category_col].fillna("Unknown").astype(str)
    d[date_col] = pd.to_datetime(d[date_col])
    d["month"] = d[date_col].dt.to_period("M").dt.to_timestamp()

    d[amount_col] = pd.to_numeric(d[amount_col], errors="coerce").fillna(0)
    d[events_col] = pd.to_numeric(d[events_col], errors="coerce").fillna(0)

    # Useful when spending values are stored as negative transactions.
    if use_abs:
        d[amount_col] = d[amount_col].abs()

    # Safe even if your dataframe is already monthly aggregated.
    g = (
        d.groupby(["month", category_col], as_index=False)
         .agg(
             amount=(amount_col, "sum"),
             events=(events_col, "sum"),
         )
    )

    g = g[g["amount"] > 0]

    months = sorted(g["month"].unique())
    if not months:
        raise ValueError("No positive spending data to plot.")

    category_order = (
        g.groupby(category_col)["amount"]
         .sum()
         .sort_values(ascending=False)
         .index
         .tolist()
    )

    palette = (
        px.colors.qualitative.Safe
        + px.colors.qualitative.Set3
        + px.colors.qualitative.Pastel
        + px.colors.qualitative.Bold
    )

    color_map = {
        category: palette[i % len(palette)]
        for i, category in enumerate(category_order)
    }

    n_months = len(months)
    ncols = min(max_cols, n_months)
    nrows = math.ceil(n_months / ncols)

    fig = go.Figure()

    for i, month in enumerate(months):
        row = i // ncols
        col = i % ncols

        # Cell position in Plotly "paper" coordinates.
        cell_w = 1 / ncols
        cell_h = 1 / nrows

        cell_x0 = col * cell_w
        cell_x1 = (col + 1) * cell_w

        # Plotly y=1 is top, y=0 is bottom.
        cell_y1 = 1 - row * cell_h
        cell_y0 = 1 - (row + 1) * cell_h

        # Reserve space below each donut for the date.
        x0 = cell_x0 + cell_w * 0.05
        x1 = cell_x1 - cell_w * 0.05
        y0 = cell_y0 + cell_h * 0.18
        y1 = cell_y1 - cell_h * 0.04

        m = g[g["month"] == month].set_index(category_col)

        values = [
            float(m.loc[c, "amount"]) if c in m.index else 0
            for c in category_order
        ]

        events = [
            int(m.loc[c, "events"]) if c in m.index else 0
            for c in category_order
        ]

        total = sum(values)
        month_label = pd.Timestamp(month).strftime("%m %Y")

        fig.add_trace(
            go.Pie(
                labels=category_order,
                values=values,
                hole=hole,
                sort=False,
                direction="clockwise",
                rotation=90,
                marker=dict(
                    colors=[color_map[c] for c in category_order],
                    line=dict(color="white", width=2),
                ),
                textinfo="label",
                textposition="inside",
                insidetextorientation="horizontal",
                customdata=events,
                hovertemplate=(
                    f"<b>%{{label}}</b><br>"
                    f"Month: {month_label}<br>"
                    "Amount: %{value:,.0f}<br>"
                    "Events: %{customdata}<br>"
                    "Share: %{percent}<extra></extra>"
                ),
                domain=dict(x=[x0, x1], y=[y0, y1]),
                showlegend=(i == 0),
            )
        )

        x_center = (x0 + x1) / 2
        y_center = (y0 + y1) / 2
        date_y = cell_y0 + cell_h * 0.07

        # Date under each donut.
        fig.add_annotation(
            x=x_center,
            y=date_y,
            xref="paper",
            yref="paper",
            text=month_label,
            showarrow=False,
            font=dict(size=13),
        )

        # Optional total amount inside donut hole.
        if show_total_in_center:
            fig.add_annotation(
                x=x_center,
                y=y_center,
                xref="paper",
                yref="paper",
                text=f"{total:,.0f}",
                showarrow=False,
                font=dict(size=14),
            )

    fig.update_layout(
        title=title,
        template="plotly_white",
        width=max(360, 250 * ncols + 180),
        height=max(330, 300 * nrows),
        margin=dict(l=20, r=170, t=70, b=40),
        legend=dict(
            title=category_col,
            x=1.02,
            y=1,
            xanchor="left",
            yanchor="top",
        ),
        uniformtext=dict(
            minsize=10,
            mode="hide",
        ),
    )

    return fig
