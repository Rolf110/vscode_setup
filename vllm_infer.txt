
from __future__ import annotations

import argparse
import asyncio
import json
import math
import os
import shutil
import tempfile
import uuid
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Sequence

import httpx
import polars as pl
from tqdm import tqdm

OpenAIMessage = dict[str, Any]
Conversation = list[OpenAIMessage]


@dataclass(frozen=True)
class RequestConfig:
    base_url: str
    api_key: str
    model: str
    temperature: float = 0.0
    top_p: float = 1.0
    max_tokens: int = 256
    timeout_s: float = 300.0
    max_concurrency: int = 64
    max_retries: int = 5
    retry_base_delay_s: float = 1.0
    retry_max_delay_s: float = 20.0
    extra_body: dict[str, Any] = field(default_factory=dict)

    @property
    def endpoint(self) -> str:
        return f"{self.base_url.rstrip('/')}/chat/completions"


@dataclass(frozen=True)
class DatasetConfig:
    input_parquet: str
    output_prefix: str
    batch_size: int
    num_parts: int
    part_index: int  # 1-based
    compression: str = "snappy"
    resume: bool = True
    cleanup_parts: bool = True


def ensure_required_columns_in_schema(schema: dict[str, Any] | pl.Schema) -> None:
    cols = set(schema.keys()) if isinstance(schema, dict) else set(schema.names())
    required = {"sample_key", "answer"}
    missing = required - cols
    if missing:
        raise ValueError(f"Input parquet is missing required columns: {sorted(missing)}")


def compute_part_bounds(total_rows: int, num_parts: int, part_index: int) -> tuple[int, int]:
    if num_parts < 1:
        raise ValueError("--num-parts must be >= 1")
    if not (1 <= part_index <= num_parts):
        raise ValueError("--part-index must be in [1, num_parts]")
    zero_based = part_index - 1
    base = total_rows // num_parts
    remainder = total_rows % num_parts
    if zero_based < remainder:
        start = zero_based * (base + 1)
        length = base + 1
    else:
        start = remainder * (base + 1) + (zero_based - remainder) * base
        length = base
    return start, length


def part_dir(output_prefix: str, part_index: int, num_parts: int) -> Path:
    return Path(f"{output_prefix}_part{part_index}of{num_parts}.parts")


def final_output_path(output_prefix: str, part_index: int, num_parts: int) -> Path:
    return Path(f"{output_prefix}_part{part_index}of{num_parts}.parquet")


def list_existing_parts(parts_dir: Path) -> list[Path]:
    if not parts_dir.exists():
        return []
    return sorted(parts_dir.glob("part_*.parquet"))


def build_conversations_from_batch(
    batch_df: pl.DataFrame,
    *,
    system_prompt: str,
) -> list[Conversation]:
    """
    Replace this function with your exact prompt formatting if needed.

    By default:
    - system = `system_prompt`
    - user = value from the `answer` column
    """
    user_messages = batch_df["answer"].to_list()
    return [
        [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": str(user_msg)},
        ]
        for user_msg in user_messages
    ]


def build_output_batch(
    batch_df: pl.DataFrame,
    generated_texts: Sequence[str | None],
    errors: Sequence[str | None],
    *,
    batch_idx: int,
    part_index: int,
    num_parts: int,
) -> pl.DataFrame:
    return batch_df.select(["sample_key", "answer"]).with_columns(
        pl.Series("model_output", list(generated_texts), dtype=pl.Utf8),
        pl.Series("error", list(errors), dtype=pl.Utf8),
        pl.lit(batch_idx).alias("batch_idx"),
        pl.lit(part_index).alias("part_index"),
        pl.lit(num_parts).alias("num_parts"),
    )


def read_batch_from_parquet(
    input_parquet: str,
    *,
    offset: int,
    length: int,
) -> pl.DataFrame:
    if length <= 0:
        return pl.DataFrame(
            {"sample_key": [], "answer": []},
            schema={"sample_key": pl.Utf8, "answer": pl.Utf8},
        )

    return (
        pl.scan_parquet(input_parquet)
        .select(["sample_key", "answer"])
        .slice(offset, length)
        .collect()
    )


async def post_chat_completion(
    client: httpx.AsyncClient,
    request_cfg: RequestConfig,
    messages: Conversation,
    *,
    request_id: str,
) -> tuple[str | None, str | None]:
    payload: dict[str, Any] = {
        "model": request_cfg.model,
        "messages": messages,
        "temperature": request_cfg.temperature,
        "top_p": request_cfg.top_p,
        "max_tokens": request_cfg.max_tokens,
    }
    payload.update(request_cfg.extra_body)

    headers = {
        "Content-Type": "application/json",
        "Authorization": f"Bearer {request_cfg.api_key}",
        "x-request-id": request_id,
    }

    attempt = 0
    while True:
        try:
            response = await client.post(
                request_cfg.endpoint,
                json=payload,
                headers=headers,
                timeout=request_cfg.timeout_s,
            )
            response.raise_for_status()
            data = response.json()
            text = data["choices"][0]["message"]["content"]
            return text, None
        except Exception as exc:
            attempt += 1
            if attempt > request_cfg.max_retries:
                return None, f"{type(exc).__name__}: {exc}"
            delay = min(
                request_cfg.retry_base_delay_s * (2 ** (attempt - 1)),
                request_cfg.retry_max_delay_s,
            )
            await asyncio.sleep(delay)


async def infer_one_with_semaphore(
    semaphore: asyncio.Semaphore,
    client: httpx.AsyncClient,
    request_cfg: RequestConfig,
    messages: Conversation,
    *,
    request_id: str,
) -> tuple[str | None, str | None]:
    async with semaphore:
        return await post_chat_completion(
            client,
            request_cfg,
            messages,
            request_id=request_id,
        )


async def infer_batch_async(
    conversations: Sequence[Conversation],
    request_cfg: RequestConfig,
    *,
    request_id_prefix: str,
) -> tuple[list[str | None], list[str | None]]:
    semaphore = asyncio.Semaphore(request_cfg.max_concurrency)

    async with httpx.AsyncClient(http2=False) as client:
        tasks = [
            infer_one_with_semaphore(
                semaphore,
                client,
                request_cfg,
                conv,
                request_id=f"{request_id_prefix}-{idx}",
            )
            for idx, conv in enumerate(conversations)
        ]
        results = await asyncio.gather(*tasks)

    texts = [text for text, _ in results]
    errors = [error for _, error in results]
    return texts, errors


def compact_parts_to_final(
    *,
    output_prefix: str,
    part_index: int,
    num_parts: int,
    sample_key_dtype: Any,
    answer_dtype: Any,
    compression: str,
    cleanup_parts: bool,
) -> Path:
    parts_path = part_dir(output_prefix, part_index, num_parts)
    final_path = final_output_path(output_prefix, part_index, num_parts)
    parts = list_existing_parts(parts_path)

    if not parts:
        empty_df = pl.DataFrame(
            schema={
                "sample_key": sample_key_dtype,
                "answer": answer_dtype,
                "model_output": pl.Utf8,
                "error": pl.Utf8,
                "batch_idx": pl.Int64,
                "part_index": pl.Int64,
                "num_parts": pl.Int64,
            }
        )
        empty_df.write_parquet(final_path, compression=compression)
        return final_path

    tmp_final = final_path.with_suffix(".tmp.parquet")
    if tmp_final.exists():
        tmp_final.unlink()

    pl.scan_parquet([str(p) for p in parts]).sink_parquet(
        str(tmp_final),
        compression=compression,
    )
    os.replace(tmp_final, final_path)

    if cleanup_parts:
        shutil.rmtree(parts_path, ignore_errors=True)

    return final_path


def run_inference_from_parquet(
    dataset_cfg: DatasetConfig,
    request_cfg: RequestConfig,
    *,
    system_prompt: str,
) -> Path:
    schema = pl.read_parquet_schema(dataset_cfg.input_parquet)
    ensure_required_columns_in_schema(schema)

    total_rows = (
        pl.scan_parquet(dataset_cfg.input_parquet)
        .select(pl.len().alias("n"))
        .collect()
        .item()
    )
    start_row, part_length = compute_part_bounds(
        total_rows,
        dataset_cfg.num_parts,
        dataset_cfg.part_index,
    )

    out_dir = final_output_path(
        dataset_cfg.output_prefix,
        dataset_cfg.part_index,
        dataset_cfg.num_parts,
    ).parent
    out_dir.mkdir(parents=True, exist_ok=True)

    parts_path = part_dir(
        dataset_cfg.output_prefix,
        dataset_cfg.part_index,
        dataset_cfg.num_parts,
    )
    final_path = final_output_path(
        dataset_cfg.output_prefix,
        dataset_cfg.part_index,
        dataset_cfg.num_parts,
    )

    if dataset_cfg.resume and final_path.exists() and not parts_path.exists():
        return final_path

    if not dataset_cfg.resume and parts_path.exists():
        shutil.rmtree(parts_path, ignore_errors=True)

    parts_path.mkdir(parents=True, exist_ok=True)

    total_batches = math.ceil(part_length / dataset_cfg.batch_size) if part_length > 0 else 0
    existing_parts = list_existing_parts(parts_path) if dataset_cfg.resume else []
    completed_batches = len(existing_parts)

    if completed_batches > total_batches:
        raise RuntimeError(
            f"Found {completed_batches} existing parts but selected dataset part has only "
            f"{total_batches} batches. Delete {parts_path} or disable resume."
        )

    progress = tqdm(
        total=total_batches,
        initial=completed_batches,
        desc=f"part {dataset_cfg.part_index}/{dataset_cfg.num_parts}",
        dynamic_ncols=True,
    )

    try:
        for batch_idx in range(completed_batches, total_batches):
            batch_start_within_part = batch_idx * dataset_cfg.batch_size
            absolute_offset = start_row + batch_start_within_part
            current_batch_len = min(
                dataset_cfg.batch_size,
                part_length - batch_start_within_part,
            )

            batch_df = read_batch_from_parquet(
                dataset_cfg.input_parquet,
                offset=absolute_offset,
                length=current_batch_len,
            )
            if batch_df.height == 0:
                break

            conversations = build_conversations_from_batch(
                batch_df,
                system_prompt=system_prompt,
            )

            request_id_prefix = (
                f"part{dataset_cfg.part_index}of{dataset_cfg.num_parts}"
                f"-batch{batch_idx:08d}-{uuid.uuid4().hex[:8]}"
            )
            texts, errors = asyncio.run(
                infer_batch_async(
                    conversations,
                    request_cfg,
                    request_id_prefix=request_id_prefix,
                )
            )

            out_batch = build_output_batch(
                batch_df,
                texts,
                errors,
                batch_idx=batch_idx,
                part_index=dataset_cfg.part_index,
                num_parts=dataset_cfg.num_parts,
            )

            part_path = parts_path / f"part_{batch_idx:08d}.parquet"
            out_batch.write_parquet(part_path, compression=dataset_cfg.compression)

            num_errors = sum(error is not None for error in errors)
            progress.update(1)
            progress.set_postfix_str(
                f"rows={min((batch_idx + 1) * dataset_cfg.batch_size, part_length)}/{part_length}, "
                f"errors={num_errors}",
                refresh=False,
            )
    finally:
        progress.close()

    final_path = compact_parts_to_final(
        output_prefix=dataset_cfg.output_prefix,
        part_index=dataset_cfg.part_index,
        num_parts=dataset_cfg.num_parts,
        sample_key_dtype=schema["sample_key"],
        answer_dtype=schema["answer"],
        compression=dataset_cfg.compression,
        cleanup_parts=dataset_cfg.cleanup_parts,
    )

    return final_path


def run_inference_from_dataframe(
    df: pl.DataFrame,
    *,
    request_cfg: RequestConfig,
    output_prefix: str,
    batch_size: int,
    num_parts: int,
    part_index: int,
    system_prompt: str,
    compression: str = "snappy",
    resume: bool = True,
    cleanup_parts: bool = True,
) -> Path:
    required = {"sample_key", "answer"}
    missing = required - set(df.columns)
    if missing:
        raise ValueError(f"Input DataFrame is missing required columns: {sorted(missing)}")

    tmp_dir = Path(tempfile.mkdtemp(prefix="vllm_async_client_"))
    tmp_input = tmp_dir / "input.parquet"
    df.write_parquet(tmp_input, compression=compression)

    try:
        dataset_cfg = DatasetConfig(
            input_parquet=str(tmp_input),
            output_prefix=output_prefix,
            batch_size=batch_size,
            num_parts=num_parts,
            part_index=part_index,
            compression=compression,
            resume=resume,
            cleanup_parts=cleanup_parts,
        )
        return run_inference_from_parquet(
            dataset_cfg,
            request_cfg,
            system_prompt=system_prompt,
        )
    finally:
        shutil.rmtree(tmp_dir, ignore_errors=True)


def parse_extra_json(raw: str | None) -> dict[str, Any]:
    if raw is None:
        return {}
    data = json.loads(raw)
    if data is None:
        return {}
    if not isinstance(data, dict):
        raise ValueError("--extra-json must decode to a JSON object or null")
    return data


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description=(
            "Async client for a vLLM OpenAI-compatible server. "
            "Reads a parquet with [sample_key, answer], processes a selected dataset part, "
            "sends one async chat-completions request per row, and writes every completed "
            "batch to parquet immediately."
        )
    )
    parser.add_argument("--input-parquet", type=str, required=True)
    parser.add_argument("--output-prefix", type=str, required=True)

    parser.add_argument("--base-url", type=str, default="http://localhost:8000/v1")
    parser.add_argument("--api-key", type=str, default=os.environ.get("VLLM_API_KEY", "EMPTY"))
    parser.add_argument("--model", type=str, required=True)

    parser.add_argument("--batch-size", type=int, default=128)
    parser.add_argument("--num-parts", type=int, default=1)
    parser.add_argument("--part-index", type=int, default=1, help="1-based part index")

    parser.add_argument("--max-concurrency", type=int, default=64)
    parser.add_argument("--timeout-s", type=float, default=300.0)
    parser.add_argument("--max-retries", type=int, default=5)
    parser.add_argument("--retry-base-delay-s", type=float, default=1.0)
    parser.add_argument("--retry-max-delay-s", type=float, default=20.0)

    parser.add_argument("--temperature", type=float, default=0.0)
    parser.add_argument("--top-p", type=float, default=1.0)
    parser.add_argument("--max-tokens", type=int, default=256)
    parser.add_argument(
        "--extra-json",
        type=str,
        default=None,
        help="Extra JSON object merged into each chat-completions request body.",
    )

    parser.add_argument("--system-prompt", type=str, default="You are concise.")
    parser.add_argument("--compression", type=str, default="snappy")
    parser.add_argument("--no-resume", action="store_true")
    parser.add_argument("--keep-parts", action="store_true")

    return parser.parse_args()


def main() -> None:
    args = parse_args()

    dataset_cfg = DatasetConfig(
        input_parquet=args.input_parquet,
        output_prefix=args.output_prefix,
        batch_size=args.batch_size,
        num_parts=args.num_parts,
        part_index=args.part_index,
        compression=args.compression,
        resume=not args.no_resume,
        cleanup_parts=not args.keep_parts,
    )

    request_cfg = RequestConfig(
        base_url=args.base_url,
        api_key=args.api_key,
        model=args.model,
        temperature=args.temperature,
        top_p=args.top_p,
        max_tokens=args.max_tokens,
        timeout_s=args.timeout_s,
        max_concurrency=args.max_concurrency,
        max_retries=args.max_retries,
        retry_base_delay_s=args.retry_base_delay_s,
        retry_max_delay_s=args.retry_max_delay_s,
        extra_body=parse_extra_json(args.extra_json),
    )

    final_path = run_inference_from_parquet(
        dataset_cfg,
        request_cfg,
        system_prompt=args.system_prompt,
    )

    print("\nFinished.")
    print(f"Output: {final_path}")


if __name__ == "__main__":
    main()
