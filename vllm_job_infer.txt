#!/usr/bin/env bash
set -euo pipefail

MODEL="meta-llama/Llama-3.1-8B-Instruct"
HOST="127.0.0.1"
PORT="8000"
API_URL="http://${HOST}:${PORT}/v1/models"
LOGFILE="vllm.log"

vllm serve "$MODEL" --host "$HOST" --port "$PORT" >"$LOGFILE" 2>&1 &
VLLM_PID=$!

cleanup() {
    echo "Cleaning up vLLM (PID $VLLM_PID)..."
    if kill -0 "$VLLM_PID" 2>/dev/null; then
        kill "$VLLM_PID" || true
        wait "$VLLM_PID" 2>/dev/null || true
    fi
}
trap cleanup EXIT INT TERM

for _ in {1..120}; do
    if curl -fsS "$API_URL" >/dev/null 2>&1; then
        echo "vLLM is ready"
        python your_script.py
        exit 0
    fi

    if ! kill -0 "$VLLM_PID" 2>/dev/null; then
        echo "vLLM crashed during startup. Last logs:"
        tail -n 50 "$LOGFILE" || true
        exit 1
    fi

    sleep 2
done

echo "Timed out waiting for vLLM. Last logs:"
tail -n 50 "$LOGFILE" || true
exit 1
